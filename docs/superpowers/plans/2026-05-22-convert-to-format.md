# Convert To Format Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a shared non-destructive `convert` audio operation with editor split-button and Browser batch support for `mp3`, `m4a`, `wav`, and `flac`, including same-extension no-op notifications.

**Architecture:** Add one import-safe format helper, one ffmpeg command builder, and one renderer used by editor and batch flows. Treat format selection as an operation parameter, not separate operations. Same-extension conversion is handled before renderer invocation so no temp file, media write, undo entry, or field replacement occurs.

**Tech Stack:** Python 3.13 add-on runtime, ffmpeg/ffprobe, Svelte 5 + TypeScript, JSON schema generated contracts, pytest, Vitest, Anki e2e tests.

```python
"""Import-safe audio output format helpers."""

from __future__ import annotations

from pathlib import Path
from typing import Any

DEFAULT_OUTPUT_FORMAT = "mp3"
SUPPORTED_OUTPUT_FORMATS = ("mp3", "m4a", "wav", "flac")

_FORMAT_LABELS = {
    "mp3": "MP3",
    "m4a": "M4A",
    "wav": "WAV",
    "flac": "FLAC",
}


def normalize_output_format(value: Any, default: str = DEFAULT_OUTPUT_FORMAT) -> str:
    """Return a supported output format for persisted configuration values."""
    if not isinstance(value, str):
        return default
    normalized = value.strip().lower()
    return normalized if normalized in SUPPORTED_OUTPUT_FORMATS else default


def validate_target_format(value: Any) -> str:
    """Return a target format for an explicit request, or raise ``ValueError``."""
    if not isinstance(value, str):
        raise ValueError(f"Unsupported output format: {value!r}")
    normalized = value.strip().lower()
    if normalized not in SUPPORTED_OUTPUT_FORMATS:
        raise ValueError(f"Unsupported output format: {value}")
    return normalized


def format_extension(target_format: Any) -> str:
    """Return the file extension, including the leading dot."""
    return f".{validate_target_format(target_format)}"


def format_label(target_format: Any) -> str:
    """Return a short user-facing format label."""
    normalized = validate_target_format(target_format)
    return _FORMAT_LABELS[normalized]


def source_matches_target_format(source_filename: str | Path, target_format: Any) -> bool:
    """Return true when the visible source filename extension already matches."""
    normalized = validate_target_format(target_format)
    return Path(str(source_filename)).suffix.lower() == f".{normalized}"
```

```python
def build_convert_audio_command(
    ffmpeg_path: Path,
    source_path: Path,
    target_format: str,
    output_path: Path,
) -> tuple[str, ...]:
    """Build an ffmpeg command that converts audio to one supported format."""
    normalized = validate_target_format(target_format)
    return (
        str(ffmpeg_path),
        "-y",
        "-i",
        str(source_path),
        "-vn",
        *_CONVERT_CODEC_ARGS[normalized],
        str(output_path),
```

```python
def render_converted_audio(
    source_path: Path,
    target_format: str,
    config: AudioProcessingConfig,
    output_path: Path | None = None,
    on_command: Callable[[tuple[str, ...]], None] | None = None,
) -> AudioProcessingResult:
    """Convert ``source_path`` to ``target_format`` without modifying audio filters."""
    normalized = validate_target_format(target_format)
    ffmpeg_path = find_ffmpeg(config.ffmpeg_path)
    if output_path is None:
        output_path = Path(tempfile.mkstemp(prefix="aqe_convert_", suffix=format_extension(normalized))[1])
    cmd = build_convert_audio_command(ffmpeg_path, source_path, normalized, output_path)
    if on_command:
        on_command(cmd)
    result = subprocess.run(
        list(cmd),
        capture_output=True,
        text=True,
        check=False,
        **_external_command_run_kwargs(),
    )  # nosec B603
    if result.returncode != 0:
        raise AudioProcessingError(result.stderr.strip() or "Audio conversion failed.")
    return AudioProcessingResult(
        output_path=output_path,
        command=cmd,
        duration_ms=probe_duration_ms(output_path, config),
    )
```

```python
def convert_async(
    editor: Any,
    command: EditorCommandPayload | None = None,
    deps: Any = None,
) -> None:
    """Convert the current media to the selected format."""
    if deps is None:
        deps = command
        command = None
    existing = deps.sessions.get(editor)
    if existing and _has_blocking_work(existing):
        deps.eval_status(editor, deps.still_processing_message, kind="processing")
        return
    config = AudioProcessingConfig.from_config(deps.config(editor))
    target_format = validate_target_format(
        command.overrides.target_format if command and command.overrides.target_format else config.output_format
    )
    session, current_path = deps.current_media_path(editor)
    if source_matches_target_format(current_path.name, target_format):
        deps.set_busy(editor, False)
        deps.eval_status(

            t("editor.status.already_target_format", {"format": format_label(target_format)}),
        )
        return
    deps.run_special_audio_transform_async(
        editor,
        label=t("editor.status.converting_format", {"format": format_label(target_format)}),
        failure_log_label="convert failed",
        renderer=lambda source_path, render_config, output_path=None, on_command=None: deps.render_converted_audio(
            source_path,
            target_format,
```

```python
def _wait_for_generated_audio(
    note,
    media_dir: Path,
    previous_name: str,
    *,
    suffix: str,
    timeout: float = 10.0,

    def current_generated() -> str | None:
        filename = _sound_filename(note.fields[0])
        if filename != previous_name and filename.endswith(suffix) and (media_dir / filename).is_file():
            return filename
        return None

    wait_for_condition(
        lambda: current_generated() is not None,
        timeout=timeout,
        message=f"Expected generated {suffix} audio reference",
    )
    generated = current_generated()
    assert generated is not None
```
